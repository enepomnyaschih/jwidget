JW.apply(jQuery.fn, {
	insert: function(item, index) {
		if (!JW.isSet(index)) {
			this.append(item);
		} else if (index == 0) {
			this.prepend(item);
		} else {
			jQuery(this.children()[index - 1]).after(item);
		}
	},
	
	replaceBy: function(el, attrs) {
		var id = attrs ? this.attr("id") : null,
			cls = attrs ? this.attr("class") : null;
		
		el = jQuery(el);
		this.after(el);
		this.detach();
		
		if (id) {
			el.attr("id", id);
		}
		if (cls) {
			el.addClass(cls);
		}
	}
});
